### React Frontend Integration

```typescript
// hooks/useCodette.ts
import { useState } from 'react';

export function useCodette() {
  const [loading, setLoading] = useState(false);
  const [response, setResponse] = useState(null);

  const query = async (queryText: string, perspectives: string[]) => {
    setLoading(true);
    try {
      const res = await fetch('/api/codette/query', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          query: queryText,
          perspectives: perspectives,
          emotion: 'curiosity'
        })
      });
      const data = await res.json();
      setResponse(data);
    } finally {
      setLoading(false);
    }
  };

  return { query, response, loading };
}

// Component usage
export function CodettePanel() {
  const { query, response, loading } = useCodette();

  return (
    <div>
      <button onClick={() => query("How do I mix vocals?", ["mix_engineering"])}>
        Ask Codette
      </button>
      {loading && <p>Codette is thinking...</p>}
      {response && (
        <div>
          {Object.entries(response.perspectives).map(([p, answer]) => (
            <div key={p}>
              <strong>{p}:</strong>
              <p>{answer}</p>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```

## ?? TROUBLESHOOTING

### Issue: "Perspective not found"
**Solution**: Use exact perspective names from Perspective enum.

### Issue: "Cocoon not created"
**Solution**: Verify emotion is valid EmotionDimension value.

### Issue: "DAW integration not connecting"
**Solution**: Ensure CodetteDAWAdapter is initialized before calling methods.

### Issue: "Slow response time"
**Solution**: Reduce number of active perspectives or increase coherence threshold.
